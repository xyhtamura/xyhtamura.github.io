/* ============================================================

============================================================ */
let slides = [];
const secRange = {};

function buildSlides() {
  slides = [];
  
  const introSec = { id: "intro", label: "Intro", type: "title" };
  secRange[introSec.id] = [0, 0];
  slides.push({ kind: "title", sec: introSec });

  PORTFOLIO.sections.forEach(sec => {
    const start = slides.length;
    slides.push({ kind: "divider", sec });
    if (sec.type === "gallery") {
      sec.pieces.forEach(piece => slides.push({ kind: "piece", sec, piece }));
    } else {
      slides.push({ kind: "grid", sec, pieces: sec.pieces });
    }
    secRange[sec.id] = [start, slides.length - 1];
  });
}
